    /********************************************
functio:uchar EncoderScan(uchar __TURN_CHANGE) 
input: 
output:
********************************************/
uchar EncoderScan(uchar __TURN_CHANGE)
{
    uchar Encod_state = 0;
    //IO口的状态
    if(Encod_RL < ESP_CHANGE) return 0;

    if(p_encoderA) {Encod_state |= 0x01;}
    if(p_encoderB) {Encod_state |= 0x02;}

    //削抖,电平削抖
    if(Encod_State_bak == Encod_state)
    {
        if(Encod_state_cnt < __TURN_CHANGE) {Encod_state_cnt++; return 0;}
    } 
    else 
    {
        Encod_State_bak = Encod_state;
        Encod_state_cnt = 0;
        return 0;
    }

    //
    Encod_State_Temp = Encod_State_Last;    //上次状态:Encod_State_Temp
    Encod_State_Last = Encod_state;         //这次状态

    if(0x03 == Encod_state) {b_Right = b_Left = 0;}

    else 
    {
        if(0x03 == Encod_State_Temp)
        {
            if(0x01 == Encod_state) {b_Right = 1;b_Left = 0;}
            else {b_Right = 0;b_Left = 1;}
        }
        else if(0x00 == Encod_State_Temp)
        {
            if(0x02 == Encod_state) {if(b_Right) {Encod_RL = 0;return 1;}}
            else if(0x01 == Encod_state) {if(b_Left) {Encod_RL = 0;return 2;}}
        }
    }
    return 0;
}
