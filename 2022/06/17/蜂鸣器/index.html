蜂鸣器

无源蜂鸣器，需要使用8K的振荡频率的脉冲驱动，才能发出声音。根据频率的大小，决定蜂鸣器叫声的尖锐程度。

在单片机中，需要先配置125us的定时器中断，蜂鸣器对应的单片机引脚，输出脉冲，即可驱动蜂鸣器

方法1

定义寄存器

uchar		Buze_Step=0;
uchar		Buze_Cnt_Time=0;
uchar		Buze_Off=0;
uchar		Buze_On=0;
uchar		Buze_Set_Cnt=0;

在125us中断中

//=========================
//位置:125us中断处
//=========================
void	Buze_Driver(void)
{
    //蜂鸣器驱动
    if(bBuze) 			
        BUZE ^= 1 ;		//异或运算，不停取反
    else	
        BUZE = 1;
}

在10ms或者20ms处

//=========================
//位置:主函数,每10ms或者20ms进来一次
//蜂鸣器控制
//=========================
void	Buze_Control(void)
{	
    //发声的时候关闭，蜂鸣器
    if (bEnable_Buze)
    {		
        ++Buze_Cnt_Time;
        if (0==Buze_Step)
        {
            if (Buze_Cnt_Time<Buze_On){bBuze=1;}			
            else
            {				
                bBuze=0;
                Buze_Cnt_Time=0;
                Buze_Step=1;
            }
        }
        else if (1==Buze_Step)
        {
            if (Buze_Cnt_Time<Buze_Off) bBuze=0;		 
            else
            {
                Buze_Step=0;
                Buze_Cnt_Time=0;
                if (Buze_Set_Cnt>0) --Buze_Set_Cnt;					
                if (0==Buze_Set_Cnt){bEnable_Buze = bSet_Buze = 0;}
            }
        }
        else{;}
    }
}

在需要蜂鸣器叫的地方，调用以下函数

//=====================================
//位置:按键处理,或者功能处,
//Buze_OnTime,发声时间;
//BuzeOffTime,毕声时间;
//Buze_Cnt,发声次数
//Buze_Set(150,0,1);
//=====================================
void	Buze_Set(u16  Buze_OnTime, u16  BuzeOffTime,u8 Buze_Cnt)
{
    if (!bSet_Buze)	   
    {
        bSet_Buze=1;
        Buze_On = Buze_OnTime/10;
        Buze_Off = BuzeOffTime/10;
        Buze_Set_Cnt = Buze_Cnt;
        Buze_Step = 0;	
        Buze_Cnt_Time=0;
        bEnable_Buze=1;		
    }	
}

/****************************************
函数名称:Beep_Pro()
输入:无
输出:无
函数功能：蜂鸣器函数 20ms/次
*****************************************/
void Beep_Pro()
{
    if(beep_count0) beep_count0--;
    else
    {
        beep_count0 = 0;
        b_beep_on = 0;
    }
}

在需要调用的地方

/****************************************
 函数名称:Beep_Inin()
输入:无
输出:无
函数功能:风扇-蜂鸣器初始化
*****************************************/
void Beep_Init(uchar _BLONG)
{
    b_beep_on = 1;
    b_fan_on = 0;
    beep_count0 = _BLONG;
}
