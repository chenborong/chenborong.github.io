void ChopWaveProcess(uchar MainMax,uchar  TarGear,uchar MinPower)
{
    uchar i,gvt_temp;
    uchar tci = 22;

    while((ad_main >= MainMax) && ( TarGear > MinPower))
    {
        if( TarGear >= tci)  TarGear--;
        tci--;
        MainMax++;
    }

    if((TarGear==0) || (TarGear>22)) {tiao_on = 0;}  //超出范围

    else
    {
        TarGear--;
        //60Hz范围内
        if(B_AC_60HZ)
        {
            for(i = 0;i < 10;i++)
            {
                if(ad_vol < VolageChopWave_60hz[i]) 
                {
                  gvt_temp = Tab_Chop220V60hz_1800W[i][TarGear];
                  break;
                }
            }
        }
        //50Hz范围内
        else 
        {
            for(i = 0;i < 10;i++)
            {
                if(ad_vol < VolageChopWave[i]) 
                {
                  gvt_temp = Tab_Chop220V_1800W[i][TarGear];
                  break;
                }
            }
        }
        tiao_on = gvt_temp;
    }
}

void JumpWaveProcess(uchar MainMax,uchar  TarGear,uchar MinPower)
{
    uchar tci = 22;

    //每超过MainMax 1个值, 档位TarGear则自减1个值,
    while((ad_main >= MainMax) && ( TarGear > MinPower))
    {
        if( TarGear >= tci)  TarGear--;
        tci--;
        MainMax++;
    }

    /********************************************
functio:void JumpWave_Ctrl(uchar  TarGear)
input: TarGear(火力)
output: b_jump_on(丢波标志位)
********************************************/
void JumpWave_Ctrl(uchar  TarGear)
{
    // uchar i;
    if((b_Scr_On || b_Scr1_On)&&(TarGear > 0))
    {    
        switch (TarGear)
        {
            case 1:
            {
                Jump_cycle = 11;
                if(0 == Jump_Cnt) b_jump_on = 1;
                else b_jump_on = 0;
                
                break;
            }
            case 2:
            {
                Jump_cycle = 21;
                if((0 == Jump_Cnt)||(8 == Jump_Cnt)||(16 == Jump_Cnt)) b_jump_on = 1;
                else b_jump_on = 0;
                
                break;
            }
            case 3:
            {
                Jump_cycle = 11;
                if((0 == Jump_Cnt)||(5 == Jump_Cnt)) b_jump_on = 1;
                else b_jump_on = 0;
                
                break;
            }
            case 4:
            {
                Jump_cycle = 21;
                if(0 == Jump_Cnt%5) b_jump_on = 1;
                else b_jump_on = 0;
            }
            case 5:
            {
                Jump_cycle = 11;
                if(0 == Jump_Cnt%4) b_jump_on = 1;
                else b_jump_on = 0;
                break;
            }
            case 6:
            {
                Jump_cycle = 21;
                if(0 == Jump_Cnt%3) b_jump_on = 1;
                else b_jump_on = 0;
                break;
            }
            case 7:
            {
                Jump_cycle = 11;
                if(0 == Jump_Cnt%3) b_jump_on = 1;
                else b_jump_on = 0;
                break;
            }
            case 8:
            {
                Jump_cycle = 21;
                if(0 == Jump_Cnt) b_jump_on = 1;
                else if((3 == Jump_Cnt)||(4 == Jump_Cnt)) b_jump_on = 1;
                else if((8 == Jump_Cnt)||(9 == Jump_Cnt)) b_jump_on = 1;
                else if((13 == Jump_Cnt)||(14 == Jump_Cnt)) b_jump_on = 1;
                else if((17 == Jump_Cnt)||(18 == Jump_Cnt)) b_jump_on = 1;
                else b_jump_on = 0;
                break;
            }
            case 9:
            {
                Jump_cycle = 11;
                if((0 == Jump_Cnt%2)&&(Jump_Cnt < 10)) b_jump_on = 1;
                else b_jump_on = 0;
                break;
            }
            case 10:
            {
                Jump_cycle = 21;
                if((0 == Jump_Cnt%4)||(1 == Jump_Cnt%4)) b_jump_on = 1;
                else b_jump_on = 0;
                break;
            }
            case 11:
            {
                Jump_cycle = 21;
                if(0 == Jump_Cnt%2) b_jump_on = 1;
                else b_jump_on = 0;
                break;
            }
            case 12:
            {
                Jump_cycle = 21;
                if(Jump_Cnt > 7)
                {
                  if(2 == Jump_Cnt%3)   b_jump_on = 0;
                  else if((3 == Jump_Cnt)||(4 == Jump_Cnt)||(7 == Jump_Cnt)) b_jump_on = 0;
                  else b_jump_on = 1;
                }
                break;
            }
            case 13:
            {
                Jump_cycle = 21;
                if(Jump_Cnt > 7)
                {
                  if(2 == Jump_Cnt%3)   b_jump_on = 0;
                  else if((4 == Jump_Cnt)||(7 == Jump_Cnt)) b_jump_on = 0;
                  else b_jump_on = 1;
                }
                break;               
            }
            case 14:
            {
                Jump_cycle = 21;
                if(Jump_Cnt > 7)
                {
                  if(2 == Jump_Cnt%3)   b_jump_on = 0;
                  else if(4 == Jump_Cnt) b_jump_on = 0;
                  else b_jump_on = 1;
                }
                break;               
            } 
            case 15:
            {
                Jump_cycle = 21;
                if(0 == Jump_Cnt%4) b_jump_on = 0;
                else b_jump_on = 1;
                break;
            }        
            case 16:
            {
                Jump_cycle = 22;
                if((Jump_Cnt > 0)&&(Jump_Cnt < 21))
                {
                    if(0 == Jump_Cnt%3) {b_jump_on = 0;break;}
                }
                b_jump_on = 1;
                break;
            }
            case 17:
            {
                Jump_cycle = 22;
                if((1 < Jump_Cnt)&&(0 == Jump_Cnt%4)) b_jump_on = 0;
                else b_jump_on = 1;
                break;
            } 
            case 18:
            {
                Jump_cycle = 22;
                if((1 < Jump_Cnt)&&(0 == Jump_Cnt%5)) b_jump_on = 0;
                else b_jump_on = 1;
                break;
            } 
            case 19:
            {
                Jump_cycle = 22;
                if((1 < Jump_Cnt)&&(0 == Jump_Cnt%6)) b_jump_on = 0;
                else b_jump_on = 1;
                break;
            }           
            case 20:
            {
                Jump_cycle = 22;
                if((1 < Jump_Cnt)&&(0 == Jump_Cnt%8)) b_jump_on = 0;
                else b_jump_on = 1;
                break;
            }
            case 21:
            {
                Jump_cycle = 22;
                if(10 == Jump_Cnt) b_jump_on = 0;
                else b_jump_on = 1;
                break;
            }
            case 22:
            {
                Jump_cycle = 22;
                b_jump_on = 1;
                break;
            }
            default:{;}
        }
        // if(Jump_Cnt < TarGear) b_jump_on = 1;
        // else b_jump_on = 0;
    }
}
