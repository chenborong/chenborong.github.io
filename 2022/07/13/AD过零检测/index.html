    AD过零检测

    1. Ad_Waitting

    在8P电陶炉中,由于电压AD采用AD检测过零点,因此与常规的AD检测方式不一样,需要在中断中,读取AD值，并处理AD值。

    void main(void)
{
    ...
    while(1)
    {
        Ad_Waitting();		//AD
        
        if(b_sys_2ms)
        {
            ...
        }
    }
}

    在主函数中,通过轮询方式检测AD，第0组，第2组为电压AD检测

    static const uchar Tab_AD_CH[4] = {
    _CH_VOL,
    _CH_PCB,
    _CH_VOL,
    _CH_MAIN
};

    void Ad_Waitting()
{
    static uchar igbtcnt,maincnt,volcnt,main_temp,vol_temp;
    static uint igbttemph,mainh,volinh;

    if(!b_Chk_Freq_ok) return;  //检测60hz

    if(b_CanRead_AD) return;    //没有读完

    // //读PCB温度和风扇AD
    if(1 == AD_SP)
    {  
        Main_inttemp >>= 5;    // 累计32次后，再右移5位
        igbttemph += Main_inttemp;
        if(igbtcnt++ >= 16)
        {
            igbtcnt = 0;
            igbttemph >>= 4;    //igbttemph <<= 4; igbttemph >>= 8;
            ad_fan = (uchar)igbttemph;
             igbttemph = 0;
        }
    }

    // //读炉面温度
    else if(3 == AD_SP)
    {
        Main_inttemp >>= 5;      // 累计32次后，再右移5位
        mainh += Main_inttemp;
        if(maincnt++ >= 16)
        {
            maincnt = 0;
            mainh >>= 4;    
            main_temp = (uchar)mainh;
            mainh = 0;

            if((main_temp < 0xf0)||(main_temp > 0x10)) {ad_main = main_temp;}
            else 
            {
                if(main_temp >= (ad_main+2)) {ad_main = main_temp - 1;}
                if(main_temp <= (ad_main-2)) {ad_main = main_temp + 1;}
            }
        }
    }

    // //读电压AD
    else 
    {
        Vol_inttemp >>= 6;
        volinh += Vol_inttemp;
        if(volcnt++ >= 48) 
        {
            volcnt = 0;
            volinh =volinh >> 5;   // volinh <<= 3; volinh >>= 8;
            vol_temp = (uchar)volinh; 
            volinh = 0;
            if(vol_temp >= (ad_vol+2)) {ad_vol = vol_temp - 1;}
            if(vol_temp <= (ad_vol-2)) {ad_vol = vol_temp + 1;}
        }
    }

    AD_HH = 0;
    if(AD_SP++ >=4) AD_SP = 0;
    if((0 == AD_SP)||(2 == AD_SP)) {b_Read_FF = 1;}
    else {b_Read_FF = 0;}	//0 和 2 为电压检测

    ADCON0 = Tab_AD_CH[AD_SP]; //选取AD通道
    ADCON1 = 0x00;              //左对齐
    asm("nop");
    asm("nop");
    asm("nop");
    asm("nop");
    AD_Cover_Go;        //启动

    CAD_JS = 0;
    b_CanRead_AD = 1;
}

    2.检查过零和AD

    在125us处，读取AD过零

    /***********************************************************************

    /********************************************
functio:void CHK_AD_READ() 
input: 
output:
********************************************/
void CHK_AD_READ()
{
    uchar inttemp;

    if(!b_Chk_Freq_ok) return;  //检测电源频率

    if(b_CanRead_AD)
    {
        if(ADCON0 & 0x02) {ADCON0 |= 0b00000010;}   //转换中
        //转换完毕
        else
        {
            if(b_Read_FF)   //检测电压
            {
               AD_HH += ADRESH;
               if(ADRESH < 4)   //AD值小于04h,零点附近
               {
                   AD_Zero++;
                   if(AD_Zero > 1)    //最低值,抖动2次，判为过零
                   {
                       if(b_Zero)       //零点位置
                       {
                           tiaojs20ms = (tiaojs >> 1);  //tiaojs20ms = tiaojs/2;
                           tiaojs = 0;
                           b_Zero = 0;
                       }
                       else {AD_Zero = 0;}
                   }
                #ifdef __JUMPWAVE_H__
                   if(tiaojs > (tiaojs20ms/3) && !b_Zero)
                   {
                       if(!b_chk_jump) {b_chk_jump = 1;}    //丢波
                   } 
                #endif//__JUMPWAVE_H__
                }
                else {AD_Zero = 0;} //ADRESH > 4
                               
                if(ADRESH > 40) //在高峰区域
                {
                    b_Zero = 1;
                    b_Zero_High = 1;
                #ifdef __JUMPWAVE_H__
                    if(!b_chk_jump) {b_chk_jump = 1;}    //丢波
                #endif
                }
                

                if(CAD_JS++ >= 72)   //20ms
                {
                    if((tiaojs == 0) && (b_Zero_High)) {b_CanRead_AD = 0;Vol_inttemp = AD_HH;}  //下一轮
                    else {AD_Cover_Go;}   //下一轮
                }
                else {AD_Cover_Go;}   //下一轮
            }
            //检测主温控，PCB的AD
            else
            {
                AD_HH += ADRESH;
                if(CAD_JS++ >= 32) { b_CanRead_AD = 0;Main_inttemp = AD_HH;} //完成
                else {AD_Cover_Go;}   //下一轮
            }
        }
    }

    tiaojs++;
    //==============================================
    //==============处理加热斩波=====================
#ifdef __CHOPWAVE_H__
    if(b_Scr_On || b_Scr1_On)
    {
        inttemp = tiaojs;
        if(inttemp > tiaojs20ms) {inttemp = inttemp - tiaojs20ms;}

        if((inttemp >= tiao_on)&&(inttemp < (tiao_on+5)))   //5*125us = 625us
        {
            if(b_Scr_On) B_SCR_MAP = 1;
            if(b_Scr1_On) B_SCR1_MAP = 1;
        }
        else {B_SCR_MAP = B_SCR1_MAP = 0;}
    }
    //关闭可控硅
    else {B_SCR_MAP = B_SCR1_MAP = 0;}
#endif

    //==============================================
    //==============处理加热丢波=====================
#ifdef __JUMPWAVE_H__
    if(b_Scr_On || b_Scr1_On)
    {
        inttemp = tiaojs;
        if(inttemp > tiaojs20ms) {inttemp = inttemp - tiaojs20ms;}

        if(Jump_Cnt >= Jump_cycle) Jump_Cnt = 0;  //20个为一个周期
 
        if((inttemp < 5)&&(b_jump_on))
        {
            if(b_Scr_On) B_SCR_MAP = 1;
            if(b_Scr1_On) B_SCR1_MAP = 1;
        }
        else {B_SCR_MAP = B_SCR1_MAP = 0;}

        //丢波
        if((inttemp == 6) && b_chk_jump)  
        {
            b_chk_jump = 0;
            Jump_Cnt++;
        }
    }
     //关闭可控硅
    else {B_SCR_MAP = B_SCR1_MAP = 0;} 
#endif
}

    对于斩波：

    只需要控制tiao_on的值，即可控制功率。tiao_on的取值范围在0~80之间，(80 * 125us = 10ms)

    对于丢波：

    只需要控制标志位b_jump_on即可加热, 不同时间段b_jump_on的状态，1为加热，0为不加热，从而达到控制功率的目的
