#ifdef __WATER_RUN__

uchar run_speed,run_state;

void dsp_water_run()
{
    if(++run_speed >= 5)
    {
        run_speed = 0;
        if(++run_state > 11) run_state = 0;
    }
    
    dspbuf0 = dspbuf1 = dspbuf2 = dspbuf3 = 0;
    
    switch(run_state)
    {
        case 0: {dspbuf0 = dspbuf1 = dspbuf2 = seg_a; break;}
        case 1: {dspbuf1 = dspbuf2 = dspbuf3 = seg_a; break;}
        case 2: {dspbuf2 = dspbuf3 = seg_a; dspbuf3+=seg_b; break;}
        case 3: {dspbuf3 = seg_a+seg_b+seg_c; break;}
        case 4: {dspbuf3 = seg_b+seg_c+seg_d; break;}
        case 5: {dspbuf3 = seg_c+seg_d; dspbuf2 = seg_d; break;}
        case 6: {dspbuf3 = seg_d; dspbuf2 = dspbuf1 = seg_d; break;}
        case 7: {dspbuf2 = dspbuf1 = dspbuf0 = seg_d; break;}
        case 8: {dspbuf1 = dspbuf0 = seg_d; dspbuf0 += seg_e; break;}
        case 9: {dspbuf0 = seg_d+seg_e+seg_f; break;}
        case 10: {dspbuf0 = seg_e+seg_f+seg_a; break;}
        case 11: {dspbuf0 = seg_f+seg_a; dspbuf1 = seg_a; break;}
        default:{}
    }
}
#endif
