    4.正反扫

    4.1清屏
    数码管,led灯全部清0；

    ;;***********************************************************
;;*NAME:_BS84B08C_DRIVR_LED_START                         							*
;;*INPUT    :                                               *
;;*OUTPUT   :                                               *
;;*USED REG.:                                               *
;;*FUNCTION :正反扫           
;;COM:	PA1 PA3 PA4 PA7 
;;SEG: PD1 PD0 PD2 PD3 PD4 PD5 PD6 PD7                                    *
;;***********************************************************				
_BS84B08C_DRIVR_LED_START:
            
            SET 	PDC
            CLR 	PD		;;SEG口设为输入,低电平
            
            SET		pc_com1
            SET		pc_com2
            SET		pc_com3
            SET		pc_com4
            
            CLR		p_com1
            CLR		p_com2
            CLR		p_com3
            CLR		p_com4	;;全部清0

    4.3扫描数码管

    scan_buf_1:
            mov 	a,led_scan_task
            addm	a,pcl
            ;;------------------;扫描数码管的4个com
            jmp 	scan_gr1_c
            jmp 	scan_gr2_c
            jmp 	scan_gr3_c
            jmp 	scan_gr4_c			
            ;;------------------;扫描数码管的4个seg			
            jmp 	scan_seg1
            jmp 	scan_seg2
            jmp 	scan_seg3
            jmp 	scan_seg4
            jmp 	scan_seg5
            jmp 	scan_seg6
            jmp 	scan_seg7
            jmp 	scan_seg8
            
scan_gr1_c:
            mov 	a,_DISP_BUF[0]
            mov 	mov_buf,a			
            clr 	pc_com1
            clr 	p_com1		;共阴数码管			
            cpla  	mov_buf		;;取反, 因为输出为0,输入为1
            jmp 	com_scan	
            
scan_gr2_c:
            mov 	a,_DISP_BUF[1]
            mov 	mov_buf,a			
            clr 	pc_com2
            clr 	p_com2		;;共阴数码管			
            cpla  	mov_buf		;;共阴数码管, 取反, 因为输出为0,输入为1
            jmp 	com_scan
            
scan_gr3_c:
            mov 	a,_DISP_BUF[2]
            mov 	mov_buf,a			
            clr 	pc_com3
            clr 	p_com3		;;共阴数码管			
            cpla  	mov_buf		;;取反, 因为输出为0,输入为1
            jmp 	com_scan
            
scan_gr4_c:
            mov 	a,_DISP_BUF[3]
            mov 	mov_buf,a			
            clr 	pc_com4
            clr 	p_com4		;;共阴数码管			
            cpla  	mov_buf		;;取反, 因为输出为0,输入为1
            jmp 	com_scan			
            

            
scan_seg1:
            clr 	acc
            sz		_DISP_BUF[4].0
            or		a,00000001b
            sz		_DISP_BUF[5].0
            or		a,00000010b
            sz		_DISP_BUF[6].0
            or		a,00000100b
            sz		_DISP_BUF[7].0

            JMP 	_DRIVR_LED_RETURN			
            
            
;;共阴扫描
com_scan:
            mov 	pdc,a		;;输入为1	输出为0
            mov		a,0ffh
            mov 	pd,a		;;高电平1	低电平0
            jmp 	scan_end	;;亮: 输出高电平
            
;;共阳扫描
seg_scan:
            mov 	a,mov_buf
            sub 	a,0
            sz		c
            jmp 	$+2
            jmp 	scan_end	;;无需点亮
            
            snz		mov_buf.0
            jmp 	$+3
            clr 	pc_com1
            set 	p_com1
            
            snz		mov_buf.1
            jmp 	$+3
            clr 	pc_com2
            set 	p_com2
            
            snz		mov_buf.2
            jmp 	$+3
            clr 	pc_com3
            set 	p_com3
            
            snz		mov_buf.3
            jmp 	$+3
            clr 	pc_com4
            set 	p_com4
            
            jmp 	scan_end	;;亮:输出低电平
