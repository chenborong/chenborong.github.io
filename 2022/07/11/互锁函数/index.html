    互锁函数

    方法1

    1个芯片，控制2个炉头。

    while((p_ch0->dsphuoli + p_ch1->dsphuoli) > _HUOLI_SUM) 	//两个机器的档位超过_HUOLI_SUM时
{
    if(b_select(MC0))  p_ch1->dsphuoli--;					//需要对另一台机器减功率
    else if(b_onoff_1) p_ch0->dsphuoli--;
}

    方法2

    1个芯片控制3个炉头，2个炉头之间互锁,3个炉头之间也要互锁

    void SINGLE_3HS_PROCESS(P_MACH_DAT p_ch0,P_MACH_DAT p_ch1,P_MACH_DAT p_ch2)
{
    uchar mac_num = 0;
    uchar _HUOLI_SUM;
    if(b_onoff) mac_num++;
    if(b_onoff_1) mac_num++;	
    if(b_onoff_2) mac_num++;
    //3个炉头的情况
    if(mac_num == 3)
    {
        _HUOLI_SUM = 12;
        while((p_ch0->dsphuoli + p_ch1->dsphuoli + p_ch2->dsphuoli)>_HUOLI_SUM)
        {
            //当前为MC0
            if(b_select(MC0))
            {
                if(p_ch1->dsphuoli > 1) p_ch1->dsphuoli--;
                else if(p_ch2->dsphuoli > 1) p_ch2->dsphuoli--;
            }
        
            //当前为MC1
            else if(b_select(MC1))
            {
                if(p_ch0->dsphuoli > 1) p_ch0->dsphuoli--;
                else if(p_ch2->dsphuoli > 1) p_ch2->dsphuoli--;
                
            }
        
            //当前为MC2
            else if(b_select(MC2))
            {
                if(p_ch0->dsphuoli > 1) p_ch0->dsphuoli--;
                else if(p_ch1->dsphuoli > 1) p_ch1->dsphuoli--;
            }
            else{;}
        }	
    }
    
    //其中两个炉工作
    else if(mac_num == 2)
    {
        _HUOLI_SUM = 12;
        if(b_onoff) _HUOLI_SUM = 13;
        
        while((p_ch0->dsphuoli + p_ch1->dsphuoli + p_ch2->dsphuoli) > _HUOLI_SUM) 
        {
                if(b_select(MC0)) 
                {
                    if(b_onoff_1) p_ch1->dsphuoli--;
                    else if(b_onoff_2) p_ch2->dsphuoli--;
                }
                
                if(b_select(MC1)) 
                {
                    if(b_onoff) p_ch0->dsphuoli--;
                    else if(b_onoff_2) p_ch2->dsphuoli--;
                }
                
                if(b_select(MC2)) 
                {
                    if(b_onoff) p_ch0->dsphuoli--;
                    else if(b_onoff_1) p_ch1->dsphuoli--;
                }
        }
    }
}
