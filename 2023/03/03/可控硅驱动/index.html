    可控硅驱动

    斩波

    volatile BIT flag_scr;
#define b_soft_zero			flag_scr.bits.Bit0
#define b_pzero_low			flag_scr.bits.Bit1 

//继电器可控硅导通相关
volatile uchar 
tiaojs,
tiao_on,
tiaojs_m,
tiao_on_m,
zero_time;

#define zero_time		80
/***********************************************
Funtion:scr_ctrl() 斩波,无须提前导通
Input:	None
Output: None
Position:125us
备注: 	控制继电器可控硅的输出
***********************************************/
void scr_ctrl()	
{	
    if(!b_chk_AC_PWR_ok)	return;		//等待检测频率完毕
    
    if(!p_zero_in)		//低电平	
    {
        if(b_pzero_low)
        {
            b_pzero_low=0;
            b_soft_zero = 1;	//
            tiaojs=0;
            
            //关机,故障,无锅,无水
            if(!b_onoff || b_error || b_nopot || !b_low_water)  
            {HEAT_OFF();}
            else
            {
                if(b_heat_en)	HEAT_ON();		//继电器过零导通与关闭
                else HEAT_OFF();
            }
        }
    }
    //高电平
    else	b_pzero_low=1;


//可控硅部分
    if(b_scr_en && !b_nopot && b_low_water)		//允许加热
    {
        if(b_soft_zero)		//低电平进来
        {
            tiaojs++;
            if((tiaojs>=tiao_on) && (tiaojs<(tiao_on+8)))				//
            {
                SCR_ON();
            }
            else if((tiaojs>=(tiao_on+zero_time)) && (tiaojs<(tiao_on+zero_time+8)))	//
            {
                SCR_ON();
            }
            else if(tiaojs >= (tiao_on+zero_time+8))
            {
                SCR_OFF();
                b_soft_zero=0;
            }
            else SCR_OFF();
        }
    }
    else SCR_OFF();
}


//控制功率
//通过修改tiao_on的值，来改变功率;
void Power_1000()
{
    tiao_on = p_1000w;	//全功率
    b_scr_en = 1;
}

void Power_100()
{
    tiao_on = p_100w;	//全功率
    b_scr_en = 1;
}

void Power_off()		//关闭可控硅
{
    b_scr_en = 0;
}

    丢波

    /*====================================================================
方式2:丢波，提前导通
====================================================================*/
#define		zero_time			76		//50hz对应半个周期的时间80    80-76 = 4 * 125 = 500us 提前了500us
#define 	Cycle_scr			20		//20个20ms为一个周期
#define 	Power_full		Cycle_scr	//全功率;
#define 	Power_1000W		10				//1000W;
#define 	Power_500W		5					//500W;
#define 	Power_100W		1					//100W;

/***********************************************
Funtion:SCR_ON(void) 
Input:	None
Output: None
Position:125us
备注: 	10个20ms为一个周期, 改变Tiao_power的值，从而改变功率
***********************************************/
void  SCR_ON(void)
{	
    if(Scr_oncycle >= 10)	Scr_oncycle = 0;	 				
    if(Scr_oncycle <= Tiao_power)
        {SCR = 1;}				
    else
        {SCR = 0;}
}

/***********************************************
Funtion:SCR_ON2(void) 
Input:	None
Output: None
Position:125us
备注: 	在一个周期的结束,判断下一个周期是否需要开通？
                若需要,则提前几us开通,
                若不需要,则不开通
***********************************************/
void SCR_ON2(void)
{
    if((Scr_oncycle+1)<=Tiao_power)		//如果下个周期需要补偿,那么允许提前导通
        {SCR = 1;}	
    else
        {SCR = 0;}
}

/***********************************************
Funtion:SCR_OFF(void) 
Input:	None
Output: None
Position:125us
备注: 	可控硅关闭状态
***********************************************/
void SCR_OFF(void)
{
    SCR = 0;
}


/***********************************************
Funtion:scr_ctrl(void) 丢波,需要提前导通
Input:	None
Output: None
Position:125us
备注: 	控制继电器可控硅的输出
***********************************************/
void  Scr_ctrl(void) 
{ 
         if(!p_zero_in)  //过零脚 == 0;
        {
            if(b_pzero_low)
             {
               b_pzero_low=0;
               bSoft_zero = 1; //
               tiaojs=0;
             }
        }
         else b_pzero_low=1;

                //可控硅部分
         if(bScr_en)  //允许加热
         {
          if(bSoft_zero)  //软件低电平进来
             {
                 tiaojs++;
               if((tiaojs>=0) && (tiaojs<8))   											 //1
                 {  SCR_ON(); }
              else if((tiaojs>=zero_time) && (tiaojs<(zero_time+8))) 	//3
              { SCR_ON();}
                       		
              else if(tiaojs >= (zero_time+80))										//5
                 { 
                     SCR_ON2();
                     bSoft_zero=0;
                   Scr_oncycle++;
                 }
                 else if(tiaojs >= (zero_time+8))											//4
               {SCR_OFF();}	
               else {SCR_OFF();}																 	//2
             }
         }
         else {SCR_OFF();}	
}

//控制功率
//Tiao_power，来改变功率;
void Power_1000()
{
    Tiao_power = p_1000w;	//全功率
    b_scr_en = 1;
}

void Power_100()
{
    Tiao_power = p_100w;	//全功率
    b_scr_en = 1;
}

void Power_off()
{
    b_scr_en = 0;
